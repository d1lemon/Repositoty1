# ============================================================
# A2A SIMULATION FOR rdrr.io/snippets
# No plumber servers
# No package installation
# ============================================================

# Load available packages
library(jsonlite)
library(randomForest)

# ============================================================
# DATA AGENT
# ============================================================

data_agent <- function() {

  data(iris)

  cat("DataAgent: Dataset loaded\n")

  list(
    sender = "DataAgent",
    receiver = "MLAgent",
    task = "train_model",
    payload = iris
  )
}

# ============================================================
# ML AGENT
# ============================================================

ml_agent <- function(message) {

  df <- as.data.frame(message$payload)

  set.seed(42)

  train_idx <- sample(
    1:nrow(df),
    size = floor(0.8 * nrow(df))
  )

  train <- df[train_idx, ]
  test <- df[-train_idx, ]

  model <- randomForest(
    Species ~ .,
    data = train
  )

  preds <- predict(model, test)

  accuracy <- mean(preds == test$Species)

  cat(
    paste(
      "MLAgent: Accuracy =",
      round(accuracy, 4),
      "\n"
    )
  )

  list(
    sender = "MLAgent",
    receiver = "ReportingAgent",
    task = "report",
    payload = list(
      accuracy = accuracy,
      rows = nrow(df),
      features = ncol(df) - 1
    )
  )
}

# ============================================================
# REPORTING AGENT
# ============================================================

reporting_agent <- function(message) {

  payload <- message$payload

  cat("\n")
  cat("====================================\n")
  cat("A2A MACHINE LEARNING REPORT\n")
  cat("====================================\n")
  cat("Rows: ", payload$rows, "\n")
  cat("Features: ", payload$features, "\n")
  cat("Accuracy: ", round(payload$accuracy, 4), "\n")
  cat("====================================\n")
}

# ============================================================
# ORCHESTRATOR
# ============================================================

cat("Starting A2A Workflow\n\n")

msg1 <- data_agent()

msg2 <- ml_agent(msg1)

reporting_agent(msg2)

cat("\nWorkflow Completed Successfully\n")
